

% Create the summ of functions

\pgfplotsset{GdbReelSum/.code={
  \ifnum\OrdrArray[#1]=1
    \edef\GdbReelSum{\GdbReelSum+GdbReelPrem(x/\WcArray[#1])}%
  \fi
  \ifnum\OrdrArray[#1]=-1
    \edef\GdbReelSum{\GdbReelSum-GdbReelPrem(x/\WcArray[#1])}%
  \fi
  \ifnum\OrdrArray[#1]=2
    \edef\GdbReelSum{\GdbReelSum+GdBReelDeux(x/\WcArray[#1],\ZArray[#1])}%
  \fi
  \ifnum\OrdrArray[#1]=-2
    \edef\GdbReelSum{\GdbReelSum-GdBReelDeux(x/\WcArray[#1],\ZArray[#1])}%
  \fi
}}
